*** Creating dataframes

#+caption: Create dataframe from a dictionary
#+begin_src python
import pandas as pd

# single key-value pairs
df = pd.DataFrame(
    {"key1": "value1", "key2": "value2"}.items(), columns=["key", "value"]
)
print(df)
#+end_src
